var SHEET_ID = 'COLE_SEU_ID_AQUI'; // Substitua pelo ID da sua planilha

function doPost(e) {
  try {
    var ss = SpreadsheetApp.openById(SHEET_ID);
    var data = JSON.parse(e.postData.contents);
    var sheet = ss.getSheetByName(data.sheet) || ss.insertSheet(data.sheet);

    if (data.action === 'formulario_feedback') {
      // Auto-create feedback from form submission
      var fbSheet = ss.getSheetByName('feedbacks') || ss.insertSheet('feedbacks');
      var fbId = 'form_' + new Date().getTime();
      var descr = 'Sentimento: ' + data.sentimento + ' | Positivos: ' + data.positivos + ' | Melhorias: ' + data.melhorias + ' | Comunicado: ' + data.comunicado + ' | Objetivos: ' + data.objetivos;
      fbSheet.appendRow([fbId, data.colaboradorId, 'feedback-colab', 'Formulário — ' + data.colaboradorNome, descr, data.data, data.satisfacao, '', '', data.colaboradorNome + ' (auto)']);
      // Also log to notificacoes sheet
      var notifSheet = ss.getSheetByName('notificacoes') || ss.insertSheet('notificacoes');
      notifSheet.appendRow(['n'+new Date().getTime(), 'Novo formulário de ' + data.colaboradorNome, 'formulario', data.colaboradorId, new Date().toISOString(), false]);
      return ContentService.createTextOutput(JSON.stringify({ok:true})).setMimeType(ContentService.MimeType.JSON);
    } else if (data.action === 'append') {
      sheet.appendRow(data.row);
    } else if (data.action === 'upsert') {
      // Upsert by ID (first column)
      var values = sheet.getDataRange().getValues();
      var found = false;
      for (var i = 1; i < values.length; i++) {
        if (String(values[i][0]) === String(data.row[0])) {
          var range = sheet.getRange(i + 1, 1, 1, data.row.length);
          range.setValues([data.row]);
          found = true; break;
        }
      }
      if (!found) sheet.appendRow(data.row);
    } else if (data.action === 'replace_all') {
      // Replace entire sheet content
      sheet.clearContents();
      if (data.headers) sheet.appendRow(data.headers);
      if (data.rows && data.rows.length) {
        sheet.getRange(2, 1, data.rows.length, data.rows[0].length).setValues(data.rows);
      }
    } else if (data.action === 'delete') {
      var vals = sheet.getDataRange().getValues();
      for (var j = vals.length - 1; j >= 1; j--) {
        if (String(vals[j][0]) === String(data.id)) {
          sheet.deleteRow(j + 1); break;
        }
      }
    }
    return ContentService.createTextOutput(JSON.stringify({ok:true}))
      .setMimeType(ContentService.MimeType.JSON);
  } catch(err) {
    return ContentService.createTextOutput(JSON.stringify({ok:false,error:err.toString()}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

function doGet(e) {
  try {
    var ss = SpreadsheetApp.openById(SHEET_ID);
    var sheetName = e.parameter.sheet;
    var sheet = ss.getSheetByName(sheetName);
    if (!sheet) return ContentService.createTextOutput('[]').setMimeType(ContentService.MimeType.JSON);
    var values = sheet.getDataRange().getValues();
    return ContentService.createTextOutput(JSON.stringify(values))
      .setMimeType(ContentService.MimeType.JSON);
  } catch(err) {
    return ContentService.createTextOutput('[]').setMimeType(ContentService.MimeType.JSON);
  }
}

// Inicializa as abas com cabeçalhos se não existirem
function enviarEmailsAcoes(emails) {
  emails.forEach(function(dest) {
    if (!dest.email || !dest.acoes || !dest.acoes.length) return;
    var linhas = dest.acoes.map(function(a) {
      return '• ' + a.acao + ' | Prazo: ' + a.prazo + ' | Status: ' + a.status + ' | Colaborador: ' + a.colaborador;
    }).join('\n');
    var corpo = 'Olá ' + dest.nome + ',\n\n' +
      'Seguem as ações sob sua responsabilidade no MF Boxer:\n\n' +
      linhas + '\n\n' +
      'Acesse o sistema para atualizar o status das ações.\n\n' +
      'Equipe MF Boxer';
    MailApp.sendEmail({
      to: dest.email,
      subject: 'Suas Acoes — MF Boxer',
      body: corpo
    });
  });
}

function inicializarPlanilha() {
  var ss = SpreadsheetApp.openById(SHEET_ID);
  var sheets = {
    'colaboradores': ['ID','Nome','Funcao','Lider','Admissao','Turno','Foto','Obs','Senha','Telefone'],
    'feedbacks':     ['ID','ColaboradorID','Tipo','Titulo','Descricao','Data','Satisfacao','Acao','Prazo','RegistradoPor'],
    'acoes':         ['ID','FeedbackID','ColaboradorID','ColaboradorNome','Acao','Prazo','Responsavel','DataConclusao','Status','Origem','CriadoEm','CriadoPor'],
    'usuarios':      ['Key','Nome','Iniciais','Senha','PodeVerTudo','Equipe'],
    'notificacoes':  ['ID','Msg','Tipo','ColaboradorID','Data','Lida'],
  };
  Object.keys(sheets).forEach(function(name) {
    var s = ss.getSheetByName(name);
    if (!s) {
      s = ss.insertSheet(name);
      s.appendRow(sheets[name]);
    }
  });
  return 'Planilha inicializada!';
}
